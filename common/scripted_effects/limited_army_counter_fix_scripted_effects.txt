# event_target:removed_army = army
decrement_titanic_life_soldier_count = {
	if = {
		limit = {
			check_variable = {
				which = titanic_life_soldier_count
				value > 0
			}
		}
		set_timed_country_flag = {
			flag = army_counter_decremented@event_target:removed_army
			days = 7
		}
		subtract_variable = {
			which = titanic_life_soldier_count
			value = 1
		}
	}
}
