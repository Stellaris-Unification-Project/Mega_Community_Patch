country_event = {
	id = primitive.10
	hide_window = yes

	trigger = {
		is_country_type = primitive
		has_country_flag = industrial_age
		NOT = { has_country_flag = recently_advanced }
	}

	mean_time_to_happen = {
		years = 100
	}

	immediate = {
		set_timed_country_flag = {
			flag = recently_advanced
			days = 2160 #60yrs
		}
		remove_country_flag = industrial_age
		set_country_flag = machine_age
		set_primitive_age = machine_age
		change_country_flag = {
			icon = {
				category = "special"
				file = "primitive.dds"
			}
			background= {
				category = "backgrounds"
				file = "new_dawn.dds"
			}
			colors={
				"yellow"
				"orange"
				"null"
				"null"
			}
		}
		#home_planet = {
		capital_scope = {
			create_army = {
				name = "NAME_Industrial_Army"
				owner = root
				species = root
				type = "industrial_army"
			}
			random_tile = {
				limit = {
					has_blocker = no
					has_building = no
					has_grown_pop = no
				}
				set_building = "building_primitive_factory"
				create_pop = {
					species = owner
				}
				clear_deposits = yes
				add_deposit = d_mineral_deposit
			}
			observation_outpost_owner = {
				country_event = { id = primitive.11 }
			}
		}
	}
}

country_event = {
	id = primitive.160
	hide_window = yes
	
	trigger = {
		is_country_type = primitive
		has_country_flag = early_space_age
		NOT = { has_country_flag = recently_advanced }
		years_passed > 25
	}

	mean_time_to_happen = {
		years = 100

		modifier = {
			factor = 0.6
			has_country_flag = acquired_tech
		}
	}

	immediate = {
		capital_scope = {
			solar_system = {
				if = {
					limit = { exists = starbase }
					starbase = {
						owner = { country_event = { id = primitive.161 } }
					}
				}
				else = {
					root = { country_event = { id = primitive.16 days = 90 } }
				}
			}
		}
	}
}

country_event = {
	id = primitive.161
	title = primitive.161.name
	desc = primitive.161.desc
	picture = GFX_evt_satellite_in_orbit
	location = from.capital_scope.solar_system

	is_triggered_only = yes

	option = {	# Stop them
		name = primitive.161.a
		allow = {
			OR = {
				custom_tooltip = {
					text = primitive.161.a.tt1
					from.capital_scope = { has_observation_outpost = yes }
				}
				custom_tooltip = {
					text = primitive.161.a.tt2
					from.capital_scope.solar_system.starbase = { has_starbase_size >= starbase_starhold }
				}
				custom_tooltip = {
					text = primitive.161.a.tt3
					from.capital_scope.solar_system = {
						any_ship_in_system = {
							is_owned_by = root
							fleet = { fleet_power > 1000 }
						}
					}
				}
				custom_tooltip = {
					text = primitive.161.a.tt4
					from.capital_scope.solar_system = {
						is_same_value = root.capital_scope.solar_system
					}
				}
			}
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0.1
				OR = {
					has_ethic = ethic_fanatic_xenophile
					has_ethic = ethic_xenophile
				}
			}
		}
		from = {
			shift_ethic = ethic_xenophobe
			hidden_effect = {
				set_timed_country_flag = {
					flag = recently_advanced
					days = 3600
				}
			}
		}
	}
	
	option = {	# Let them
		name = primitive.161.b
		trigger = {
			NOT = {
				from.capital_scope.solar_system = {
					is_same_value = root.capital_scope.solar_system
				}
			}
		}
		ai_chance = {
			factor = 10
		}
		from = {
			hidden_effect = {
				country_event = { id = primitive.16 days = 90 }
			}
		}
	}
}

			AND = {
				exists = from
				from = {
					is_country_type_with_subjects = yes
				}
			}
			AND = {
				NOT = { has_country_flag = recently_advanced }
				years_passed > 25
