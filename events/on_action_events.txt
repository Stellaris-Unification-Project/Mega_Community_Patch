	id = action.999
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			is_country_type = default
			is_fallen_empire = yes
		}
		fromfrom = {
			any_owned_pop = {
				is_being_force_purged = yes
			}
		}
	}
	immediate = {
		fromfrom = {
			every_owned_pop = {
				limit = {
					is_being_force_purged = yes
					#NOT = { has_citizenship_type = { country = root type = citizenship_purge } }
					#NOT = { has_citizenship_type = { country = root type = citizenship_purge_machine } }
				}
				purge = no
			}
		}
	}
}
